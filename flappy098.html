Flappy bird using arduino uno and .9" lcd i2c

üìñ Project description

This project is a miniature version of the popular Flappy Bird game, recreated using an Arduino Uno and a 0.96-inch I¬≤C OLED display. The goal is to demonstrate how classic game mechanics can be implemented on low-cost microcontrollers. Players control a small bird that automatically falls due to gravity. By pressing a button, the bird "flaps" upward to avoid obstacles (pipes) moving from right to left across the screen. The challenge is to navigate through the gaps in the pipes without colliding. Each successful pass through a pipe increases the score.

#include 
#include 
#include 

#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define BUTTON_PIN A0  
#define BIRD_X 20  
#define PIPE_WIDTH 20  
#define GAP_HEIGHT 30  
#define GRAVITY 1  
#define JUMP_STRENGTH -6  

Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, -1);

int birdY = SCREEN_HEIGHT / 2; 
int birdVelocity = 0;
int pipeX = SCREEN_WIDTH;  
int pipeGapY;  
bool gameOver = false;
bool gameStarted = false;
int highestScore = 0;
int score = 0;

void setup() {
  pinMode(BUTTON_PIN, INPUT_PULLUP); 
  Serial.begin(115200);

  Wire.begin(SDA_PIN, SCL_PIN);  

  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println(F("SSD1306 not found!"));
    for (;;);
  }

  randomSeed(analogRead(0));  
  pipeGapY = random(10, SCREEN_HEIGHT - GAP_HEIGHT - 10);
  displayHomeScreen();
}

void loop() {
  if (!gameStarted) {
    if (digitalRead(BUTTON_PIN) == LOW) {
      gameStarted = true;
      restartGame();
    }
    return;
  }

  if (gameOver) {
    displayGameOver();
    if (digitalRead(BUTTON_PIN) == LOW) {
      displayHomeScreen();
      gameStarted = false;
    }
    return;
  }

  birdVelocity += GRAVITY;
  birdY += birdVelocity;

  if (digitalRead(BUTTON_PIN) == LOW) {
    birdVelocity = JUMP_STRENGTH;
  }

  pipeX -= 2;
  if (pipeX < -PIPE_WIDTH) {
    pipeX = SCREEN_WIDTH;
    pipeGapY = random(10, SCREEN_HEIGHT - GAP_HEIGHT - 10);
    score++;
    if (score > highestScore) {
      highestScore = score;
    }
  }

  if (birdY <= 0 || birdY >= SCREEN_HEIGHT ||
      (BIRD_X + 5 > pipeX && BIRD_X < pipeX + PIPE_WIDTH &&
       (birdY < pipeGapY || birdY > pipeGapY + GAP_HEIGHT))) {
    gameOver = true;
  }

  drawGame();
  delay(30);
  }

  void drawGame() {
  display.clearDisplay();

    for (int i = 0; i < SCREEN_WIDTH; i += 16) {
    display.fillTriangle(i, SCREEN_HEIGHT, i + 8, SCREEN_HEIGHT - 10, i + 16, SCREEN_HEIGHT, SSD1306_WHITE);
  }

  display.fillRect(BIRD_X, birdY, 6, 4, SSD1306_WHITE); 
  display.drawPixel(BIRD_X + 6, birdY + 1, SSD1306_WHITE); 

  display.fillRoundRect(pipeX, 0, PIPE_WIDTH, pipeGapY, 5, SSD1306_WHITE);
  display.fillRoundRect(pipeX, pipeGapY + GAP_HEIGHT, PIPE_WIDTH, SCREEN_HEIGHT, 5, SSD1306_WHITE);

  display.display();
 }

  void displayGameOver() {
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(40, 20);
  display.println("Game Over");
  display.setCursor(30, 30);
  display.println("Press Button");
  display.display();
 }

  void displayHomeScreen() {
  display.clearDisplay();
  display.setTextSize(2);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(5, 10);
  display.println("FLAPPY RIC");
  display.setTextSize(1);
  display.setCursor(20, 40);
  display.print("High Score: ");
  display.print(highestScore);
  display.setCursor(10, 50);
  display.println("Press Btn to Start");
  display.display();
 }

  void restartGame() {
  birdY = SCREEN_HEIGHT / 2;
  birdVelocity = 0;
  pipeX = SCREEN_WIDTH;
  pipeGapY = random(10, SCREEN_HEIGHT - GAP_HEIGHT - 10);
  gameOver = false;
  score = 0;
 }
